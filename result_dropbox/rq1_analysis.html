One Sample Z-Test for a Proportion

by_type <- data %>% 
  group_by(type) %>%
  summarise(Count = n())

by_type <- by_type %>%
  mutate(Prop = Count / sum(Count))

print(by_type)
## # A tibble: 2  3

##   <chr>  <int> <dbl>

test_result <- prop.test(x = by_type$Count[2], n = sum(by_type$Count), p = 0.5, correct = FALSE)

##

by_region <- data %>%
